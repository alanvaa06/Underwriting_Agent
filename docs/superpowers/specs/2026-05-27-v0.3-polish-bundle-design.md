# v0.3 Polish Bundle — 5 Features

**Date:** 2026-05-27
**Status:** Awaiting user review

## 1. Goal

Add 5 portfolio-grade polish features in a single shippable release:

1. **Cost meter** — per-agent USD breakdown in decision card
2. **Decision PDF export** — client-side jsPDF download button
3. **Dark mode** — toggle button, persisted, `prefers-color-scheme` initial
4. **Live token streaming** — per-agent tokens stream via LangChain async callbacks

## 3. Cost Meter (§1)

### 3.1 Backend

**`underwriter/pricing.py`** — NEW:

```python
"""OpenAI gpt-4o pricing constants + per-usage cost calc."""

# Per-token USD as of 2026-05. Update when OpenAI changes pricing.
GPT_4O_INPUT_USD_PER_TOKEN = 2.50 / 1_000_000
GPT_4O_OUTPUT_USD_PER_TOKEN = 10.00 / 1_000_000


def compute_cost(usage: dict) -> float:
    """usage = {'input_tokens': int, 'output_tokens': int}"""
    return (
        usage.get("input_tokens", 0) * GPT_4O_INPUT_USD_PER_TOKEN
        + usage.get("output_tokens", 0) * GPT_4O_OUTPUT_USD_PER_TOKEN
    )
```

**`underwriter/agents/base.py`** — `invoke_agent` signature changes:

```python
async def invoke_agent(llm, *, system_prompt, user_prompt) -> tuple[dict, dict]:
    """Returns (parsed_json, usage_dict). usage_dict has input_tokens, output_tokens."""
    msg = await llm.ainvoke([SystemMessage(content=system_prompt), HumanMessage(content=user_prompt)])
    # ... parse JSON as before ...
    usage = getattr(msg, "usage_metadata", None) or {}
    usage_norm = {
        "input_tokens": usage.get("input_tokens", 0),
        "output_tokens": usage.get("output_tokens", 0),
    }
    return parsed, usage_norm
```

Each agent node returns `usage` in state delta:

```python
return {
    "credit_analysis": json.dumps(result),
    "reasoning_chain": [...],
    "usage": {"credit": usage_norm},  # accumulates via reducer
}
```

**`underwriter/state.py`** — add field:

```python
class UnderwritingState(TypedDict):
    # ... existing ...
    usage: Annotated[dict[str, dict], _merge_usage]  # agent_name → {input_tokens, output_tokens}
```

With `_merge_usage(left, right) → {**left, **right}` reducer (importable from state.py).

(function () {
  function exportDecisionPDF(state) {
    const { jsPDF } = window.jspdf;
    const doc = new jsPDF({ unit: 'pt', format: 'a4' });
    let y = 40;

    // Header
    doc.setFontSize(18).setFont(undefined, 'bold');
    doc.text('Underwriter Agent — Decision Memo', 40, y);
    y += 24;
    doc.setFontSize(9).setFont(undefined, 'normal').setTextColor(120);
    doc.text(`Generated ${new Date().toLocaleString()}`, 40, y);
    y += 28;

    // Decision banner
    const colors = { APPROVED: [16,185,129], CONDITIONAL_APPROVAL: [245,158,11], DENIED: [239,68,68] };
    const c = colors[state.decision] || [128,128,128];
    doc.setFillColor(...c); doc.rect(40, y, 515, 36, 'F');
    doc.setTextColor(255).setFontSize(18).setFont(undefined, 'bold');
    doc.text(state.decision.replace('_',' '), 50, y + 24);
    doc.text(`Risk ${state.risk_score}/100`, 460, y + 24);
    y += 56;

    // Memo
    doc.setTextColor(40).setFontSize(11).setFont(undefined, 'normal');
    const memoLines = doc.splitTextToSize(state.memo || '', 515);
    doc.text(memoLines, 40, y);
    y += memoLines.length * 14 + 16;

    // Per-agent summaries
    doc.setFont(undefined, 'bold').setFontSize(12);
    doc.text('Agent Findings', 40, y); y += 18;
    doc.setFont(undefined, 'normal').setFontSize(10);
    for (const [name, output] of Object.entries(state.agent_outputs || {})) {
      const summary = (output && output[`${name}_analysis`])
        ? (JSON.parse(output[`${name}_analysis`]).summary || '')
        : '';
      doc.setFont(undefined, 'bold').text(name.toUpperCase(), 40, y);
      doc.setFont(undefined, 'normal');
      const lines = doc.splitTextToSize(summary, 470);
      doc.text(lines, 95, y);
      y += Math.max(14, lines.length * 12) + 6;
    }

    // Cost footer
    if (state.cost) {
      y += 10;
      doc.setFontSize(9).setTextColor(120);
      doc.text(`Total cost: $${state.cost.total_usd.toFixed(4)} (${state.cost.total_tokens || ''} tokens)`, 40, y);
    }

    // Branding footer
    doc.setFontSize(8).setTextColor(150);
    doc.text('Generated by github.com/alanvaa06/Underwriting_Agent', 40, 820);

    doc.save(`underwriting-decision-${Date.now()}.pdf`);
  }

  window.UnderwriterPDF = { exportDecisionPDF };
})();
```

(function () {
  const root = document.documentElement;

  function apply(theme) {
    if (theme === 'dark') root.classList.add('dark');
    else root.classList.remove('dark');
    if (window.UnderwriterGraph) window.UnderwriterGraph.setTheme?.(theme);
  }

  function initial() {
    const saved = localStorage.getItem('uw-theme');
    if (saved) return saved;
    return window.matchMedia('(prefers-color-scheme: dark)').matches ? 'dark' : 'light';
  }

  function toggle() {
    const current = root.classList.contains('dark') ? 'dark' : 'light';
    const next = current === 'dark' ? 'light' : 'dark';
    localStorage.setItem('uw-theme', next);
    apply(next);
  }

  apply(initial());
  document.addEventListener('DOMContentLoaded', () => {
    document.getElementById('dark-toggle')?.addEventListener('click', toggle);
  });

  window.UnderwriterDark = { apply, toggle };
})();
```

## 6. Live Token Streaming (§4)

### 6.1 Approach

LangChain `AsyncCallbackHandler` subclass receives `on_llm_new_token(token)` callbacks. Each per-agent LLM call gets its own callback wrapper that pushes tokens onto a shared `asyncio.Queue`. `stream_run` drains the queue concurrently with `graph.astream` and yields `token` AgentEvents.

### 6.2 `build_llm` change

```python
def build_llm(*, api_key: str, model: str = "gpt-4o") -> BaseChatModel:
    return ChatOpenAI(api_key=api_key, model=model, temperature=0.1, timeout=60, streaming=True)
```

### 6.3 Callback handler — added to `underwriter/streaming.py`

```python
import asyncio
from langchain_core.callbacks.base import AsyncCallbackHandler


def make_token_callback(agent_name: str, queue: asyncio.Queue) -> AsyncCallbackHandler:
    class _Cb(AsyncCallbackHandler):
        async def on_llm_new_token(self, token: str, **kwargs):
            await queue.put(AgentEvent(
                type="token",
                payload={"agent": agent_name, "token": token},
                ts=time.time(),
            ))
    return _Cb()
```

### 6.4 `build_workflow` accepts callback factory

```python
def build_workflow(*, llm, retriever, callback_factory=None):
    def with_cb(agent_name, base_llm):
        if callback_factory is None:
            return base_llm
        return base_llm.with_config(callbacks=[callback_factory(agent_name)])

    async def credit_node(s):
        return await credit_analyst_node(s, llm=with_cb("credit", llm), retriever=retriever)
    # ... same pattern for income, asset, collateral, critic, decision ...
```

### 6.5 `stream_run` concurrent drain

```python
async def stream_run(*, applicant, llm, retriever, case_id=None):
    queue: asyncio.Queue = asyncio.Queue()
    graph = build_workflow(
        llm=llm, retriever=retriever,
        callback_factory=lambda name: make_token_callback(name, queue),
    )
    # ... init state ...

    async def run_graph():
        nonlocal final_state
        async for chunk in graph.astream(state, config=config, stream_mode="updates"):
            for node_name, delta in chunk.items():
                if node_name not in _INTERNAL_NODES:
                    await queue.put(AgentEvent(type="agent_start", payload={"agent": node_name}, ts=time.time()))
                    await queue.put(AgentEvent(type="agent_complete", payload={"agent": node_name, "output": _serializable(delta)}, ts=time.time()))
                final_state.update(delta)
        await queue.put(None)  # sentinel

    asyncio.create_task(run_graph())
    while True:
        evt = await queue.get()
        if evt is None: break
        yield evt

    yield AgentEvent(type="decision", payload={...}, ts=time.time())
    yield AgentEvent(type="cost", payload=_cost_payload(final_state.get("usage", {})), ts=time.time())
    yield AgentEvent(type="done", payload={...}, ts=time.time())
```

### 6.6 Agent functions go async

All 6 agent node functions become `async def` and use `await llm.ainvoke(...)` via the updated `invoke_agent`.
